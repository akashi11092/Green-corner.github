function getSpreadsheet() {
  try {
    if (SPREADSHEET_ID && SPREADSHEET_ID !== '') {
      return SpreadsheetApp.openById(SPREADSHEET_ID);
    } else {
      var ss = SpreadsheetApp.getActiveSpreadsheet();
      if (!ss) {
        ss = SpreadsheetApp.create('Зеленый уголок - Магазин');
      }
      return ss;
    }
  } catch (e) {
    Logger.log('Ошибка таблицы: ' + e.toString());
    return SpreadsheetApp.create('Зеленый уголок - Магазин');
  }
}

function registerUser(userData) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Пользователи');
    
    if (!sheet) {
      sheet = ss.insertSheet('Пользователи');
      sheet.appendRow(['ID', 'Дата', 'ФИО', 'Телефон', 'Email', 'Карта', 'Статус']);
    }
    
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][3] == userData.phone) {
        return { success: false, message: 'Пользователь уже зарегистрирован' };
      }
    }
    
    var userId = 'USER_' + new Date().getTime();
    var cardLast4 = userData.card ? userData.card.replace(/\s/g, '').slice(-4) : '****';
    
    // Сохраняем телефон КАК ЕСТЬ (без апострофа)
    sheet.appendRow([userId, new Date(), userData.name, userData.phone, userData.email || '', cardLast4, 'Активен']);
    
    return { success: true, message: 'Регистрация успешна!', userId: userId, name: userData.name };
  } catch (e) {
    Logger.log('Ошибка регистрации: ' + e.toString());
    return { success: false, message: 'Ошибка: ' + e.toString() };
  }
}

function loginUser(phone) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Пользователи');
    
    if (!sheet) {
      return { success: false, message: 'Сначала зарегистрируйтесь' };
    }
    
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][3] == phone) {
        return { success: true, userId: data[i][0], name: data[i][2], phone: data[i][3], email: data[i][4] || '' };
      }
    }
    
    return { success: false, message: 'Пользователь не найден' };
  } catch (e) {
    Logger.log('Ошибка входа: ' + e.toString());
    return { success: false, message: 'Ошибка: ' + e.toString() };
  }
}

function processOrder(orderData) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Заказы');
    
    if (!sheet) {
      sheet = ss.insertSheet('Заказы');
      sheet.appendRow(['ID', 'Дата', 'UserID', 'Имя', 'Телефон', 'Email', 'Адрес', 'Товары', 'Сумма', 'Оплата', 'Статус', 'Дата доставки']);
    }
    
    var orderId = 'ORD_' + new Date().getTime();
    var itemsText = orderData.items.map(function(item) {
      return item.name + '|' + item.quantity + '|' + item.price + '|' + item.image;
    }).join(';');
    
    var deliveryDate = new Date();
    deliveryDate.setDate(deliveryDate.getDate() + 2);
    
    // ВАЖНО: Сохраняем телефон как текст с префиксом ' чтобы Google Sheets не ломал
    var phoneValue = "'" + orderData.phone;
    
    sheet.appendRow([
      orderId, 
      new Date(), 
      orderData.userId || '', 
      orderData.name, 
      phoneValue,  // <-- ТЕПЕРЬ ПРАВИЛЬНО
      orderData.email, 
      orderData.address, 
      itemsText, 
      orderData.total, 
      orderData.paymentMethod, 
      'Оформлен', 
      deliveryDate
    ]);
    
    Logger.log('✅ Заказ создан: ' + orderId + ' для телефона: ' + orderData.phone);
    
    return { success: true, message: 'Заказ оформлен!', orderId: orderId };
  } catch (e) {
    Logger.log('❌ Ошибка заказа: ' + e.toString());
    return { success: false, message: 'Ошибка при оформлении: ' + e.toString() };
  }
}

function getUserOrders(userId, userPhone) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName('Заказы');
    
    Logger.log('🔍 Поиск заказов для userId: ' + userId + ', phone: ' + userPhone);
    
    if (!sheet) {
      Logger.log('❌ Лист Заказы не найден');
      return { success: true, orders: [] };
    }
    
    var data = sheet.getDataRange().getValues();
    var orders = [];
    
    Logger.log('📊 Всего записей в таблице: ' + data.length);
    
    for (var i = 1; i < data.length; i++) {
      var orderUserId = data[i][2] || '';
      var orderPhone = data[i][4] || '';
      
      // Убираем апостроф из телефона для сравнения
      var cleanOrderPhone = orderPhone.toString().replace(/^'/, '').trim();
      var cleanUserPhone = userPhone.toString().trim();
      
      Logger.log('📱 Проверка заказа #' + i + ': userId=' + orderUserId + ', orderPhone=' + cleanOrderPhone + ', userPhone=' + cleanUserPhone);
      
      // Ищем по userId ИЛИ по телефону
      if (orderUserId == userId || cleanOrderPhone == cleanUserPhone) {
        Logger.log('✅ Заказ найден!');
        
        var itemsArray = (data[i][7] || '').split(';');
        var items = itemsArray.map(function(item) {
          var parts = item.split('|');
          return { 
            name: parts[0] || 'Товар', 
            quantity: parts[1] || '1', 
            price: parts[2] || '0', 
            image: parts[3] || '' 
          };
        });
        
        orders.push({
          id: data[i][0],
          date: data[i][1],
          name: data[i][3],
          phone: cleanOrderPhone,
          address: data[i][6],
          items: items,
          total: data[i][8],
          paymentMethod: data[i][9],
          status: data[i][10] || 'Оформлен',
          deliveryDate: data[i][11]
        });
      }
    }
    
    Logger.log('🎉 Найдено заказов: ' + orders.length);
    
    // Сортируем по дате (новые сверху)
    orders.sort(function(a, b) { return new Date(b.date) - new Date(a.date); });
    
    return { success: true, orders: orders };
  } catch (e) {
    Logger.log('❌ Ошибка получения заказов: ' + e.toString());
    return { success: false, orders: [], error: e.toString() };
  }
}

function displayOrders(response) {
  var container = document.getElementById('ordersList');
  
  if (!response.success) {
    container.innerHTML = 
      '<div style="text-align:center;padding:2rem;color:var(--wb-red)">❌ Ошибка: ' + (response.error || response.message) + '</div>';
    return;
  }
  
  if (!response.orders || response.orders.length === 0) {
    container.innerHTML = 
      '<div style="text-align:center;padding:3rem;color:var(--light-text)">' +
      '<p style="font-size:1.2rem;margin-bottom:1rem">📭 У вас пока нет заказов</p>' +
      '<button class="btn" onclick="showPage(\'catalog\')">Перейти в каталог</button>' +
      '</div>';
    return;
  }
  
  console.log('✅ Найдено заказов:', response.orders.length);
  
  var html = '';
  response.orders.forEach(function(order) {
    var statusClass = 'status-' + order.status.replace(/\s/g, '-');
    var orderDate = new Date(order.date);
    var day = orderDate.getDate();
    var month = orderDate.toLocaleString('ru-RU', { month: 'short' }).toUpperCase();
    var isPaid = order.paymentMethod === 'Онлайн-оплата';
    
    html += '<div class="order-card">';
    
    // Header
    html += '<div class="order-card-header">';
    html += '<div class="order-date"><span class="order-date-day">' + day + '</span>' + month + '</div>';
    html += '<span class="order-status-badge ' + statusClass + '">' + order.status + '</span>';
    html += '</div>';
    
    // Items
    html += '<div class="order-card-body">';
    order.items.forEach(function(item) {
      html += '<div class="order-item-card">';
      html += '<div class="order-item-image-container">';
      html += '<img src="' + (item.image || 'https://via.placeholder.com/120') + '" class="order-item-image" alt="' + item.name + '">';
      if (!isPaid) {
        html += '<div class="unpaid-badge">НЕ ОПЛАЧЕН</div>';
      }
      html += '</div>';
      html += '<div class="order-item-details">';
      html += '<div><div class="order-item-name">Гипоаллергенным</div><div class="order-item-title">' + item.name + '</div></div>';
      html += '<div class="order-item-quantity">' + item.quantity + ' шт.</div>';
      html += '<div class="order-item-price">' + parseInt(item.price).toLocaleString() + ' ₽</div>';
      html += '</div></div>';
    });
    html += '</div>';
    
    // Tracker
    html += '<div class="order-tracker">';
    html += '<div class="tracker-step completed"><div class="tracker-circle">✓</div><div class="tracker-label">Оформлен</div></div>';
    html += '<div class="tracker-step ' + (order.status === 'В пути' || order.status === 'Приехал' || order.status === 'Доставлен' ? 'completed' : 'active') + '"><div class="tracker-circle">🚚</div><div class="tracker-label">В пути</div></div>';
    html += '<div class="tracker-step ' + (order.status === 'Приехал' || order.status === 'Доставлен' ? 'completed' : '') + '"><div class="tracker-circle">🏠</div><div class="tracker-label">Приехал</div></div>';
    html += '</div>';
    
    // Footer
    html += '<div class="order-card-footer">';
    html += '<div class="order-total-block">';
    html += '<div class="order-total-label">К оплате</div>';
    html += '<div class="order-total-amount">' + order.total + ' ₽</div>';
    html += '</div>';
    html += '<button class="order-payment-btn ' + (isPaid ? 'paid' : '') + '">' + (isPaid ? '✓ Оплачено' : 'К оплате ' + order.total + ' ₽') + '</button>';
    html += '</div>';
    
    html += '</div>';
  });
  
  container.innerHTML = html;
}
